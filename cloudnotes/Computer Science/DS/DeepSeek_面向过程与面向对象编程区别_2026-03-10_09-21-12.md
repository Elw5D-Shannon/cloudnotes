# order_processing.py - 面向过程风格的订单处理

# 全局数据结构
orders_db = []  # 订单数据库

def create_user(name, email, address):

    user = {
        'id': len(users_db) + 1,

        'email': email,
        'address': address,

def create_product(name, price, stock):
    """创建商品"""
    product = {
        'id': len(products_db) + 1,
        'name': name,
        'price': price,
        'stock': stock
    }
    products_db.append(product)
    inventory[product['id']] = stock
    return product

def check_stock(product_id, quantity):
    """检查库存"""
    return inventory.get(product_id, 0) >= quantity

def update_stock(product_id, quantity):
    """更新库存"""
    if check_stock(product_id, quantity):
        inventory[product_id] -= quantity
        return True
    return False

def calculate_total(products):
    """计算订单总价"""
    total = 0
    for product in products:
        total += product['price'] * product['quantity']
    return total

def create_order(user_id, product_list):
    """创建订单"""
    # 验证用户
    user = next((u for u in users_db if u['id'] == user_id), None)
    if not user:
        return {"error": "User not found"}
    
    # 验证库存
    for item in product_list:
        if not check_stock(item['product_id'], item['quantity']):
            return {"error": f"Insufficient stock for product {item['product_id']}"}
    
    # 计算总价
    total = 0
    order_products = []
    for item in product_list:
        product = next(p for p in products_db if p['id'] == item['product_id'])
        subtotal = product['price'] * item['quantity']
        order_products.append({
            'product_id': item['product_id'],
            'product_name': product['name'],
            'quantity': item['quantity'],
            'unit_price': product['price'],
            'subtotal': subtotal
        })
        total += subtotal
    
    # 创建订单
    order = {
        'id': len(orders_db) + 1,
        'user_id': user_id,
        'products': order_products,
        'total_amount': total,
        'status': 'pending',
        'created_at': '2024-01-01 10:00:00'
    }
    
    orders_db.append(order)
    user['orders'].append(order['id'])
    
    # 更新库存
    for item in product_list:
        update_stock(item['product_id'], item['quantity'])
    
    return order

def process_payment(order_id, payment_method):
    """处理支付"""
    order = next((o for o in orders_db if o['id'] == order_id), None)
    if not order:
        return {"error": "Order not found"}
    
    if order['status'] != 'pending':
        return {"error": "Order already processed"}
    
    # 模拟支付处理
    payment_result = {
        'order_id': order_id,
        'method': payment_method,
        'amount': order['total_amount'],
        'status': 'success',
        'transaction_id': 'TXN123456'
    }
    
    order['status'] = 'paid'
    order['payment'] = payment_result
    
    return payment_result

def ship_order(order_id, tracking_number):
    """发货"""
    order = next((o for o in orders_db if o['status'] == 'paid'), None)
    if order:
        order['status'] = 'shipped'
        order['tracking_number'] = tracking_number
        return order
    return {"error": "Order not ready for shipping"}

## 四、优缺点对比

### 面向过程（Python）优点：
1. **开发速度快**：适合快速原型开发
2. **代码直观**：逻辑流程清晰，易于理解
3. **灵活性高**：可以随时添加新函数

### 面向过程缺点：
1. **维护困难**：全局变量容易被意外修改
2. **扩展性差**：添加新功能需要修改多处代码
3. **缺乏封装**：数据和操作分离，难以保证数据一致性

### 面向对象（C++）优点：
1. **维护性好**：封装性好，修改不影响外部代码
2. **扩展性强**：通过继承和多态容易添加新功能
3. **模块化程度高**：每个类职责明确，便于团队协作

### 面向对象缺点：
1. **开发复杂度高**：需要设计类层次结构
2. **学习曲线陡**：需要理解面向对象设计原则
3. **可能过度设计**：简单功能可能过度抽象

## 五、大型项目选择建议

**适合面向过程的场景：**
- 小型到中型项目
- 数据处理流程固定的应用

**适合面向对象的场景：**
- 大型复杂系统
- 需要长期维护的项目
- 多团队协作开发
- 业务逻辑复杂多变的系统
