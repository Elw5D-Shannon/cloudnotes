
给定一个图片路径、长度为 3 的二进制数和一个像素的坐标点，将该二进制数的每一位从高位开始（从二进制数左侧）依次写入给定坐标的像素的 r，g，b 三个值，最后保存图片。
将给定消息依次写入每个像素的r,g,b三个值的最低有效位，像素写入顺序为按像素列读取，如下：（1,1）、（1,2）……（1，height）、（2,1）、（2,2）……（2，height）……（width，height）
```python
from PIL import Image
import os
########## Begin ##########
def embed_message(image_path, message, output_path):
    """在RGB图像的最低有效位中嵌入消息"""
    image_path = os.path.join("/data/workspace/myshixun/src/step3",image_path)
    output_path = os.path.join("/data/workspace/myshixun/src/step3",output_path)
    # 加载图像
    img = Image.open(image_path)
    if img.mode != 'RGB':
        img = img.convert('RGB')
    # 获取图像的pixels
    pixels = img.load()
    #获取图像的宽、高
    width, height = img.size
    # 将消息转换为二进制并加上终止符'1111111111111110'
    binary_message = ''.join(format(ord(char), '08b') for char in message) + '1111111111111110'
    #判断图像的大小是否满足消息的写入，及消息的二进制格式长度小于图像width*height*3,不满足则打印失败警告
    if len(binary_message) > width * height * 3:
                print("Fail! The message is too long！")
    else:
        # 嵌入消息到图像的RGB通道
        message_index = 0
        message_length = len(binary_message)
        # 按列遍历像素：先列后行
        for x in range(width):
            for y in range(height):
                r, g, b = pixels[x, y]
                # 修改R通道的LSB
                if message_index < message_length:
                    # 清除最低有效位并设置新的LSB
                    r = (r & 0xFE) | int(binary_message[message_index])
                    message_index += 1
                # 修改G通道的LSB
                if message_index < message_length:
                    g = (g & 0xFE) | int(binary_message[message_index])
                    message_index += 1
                # 修改B通道的LSB
                if message_index < message_length:
                    b = (b & 0xFE) | int(binary_message[message_index])
                    message_index += 1
                # 更新像素值
                pixels[x, y] = (r, g, b)
                # 如果消息已经全部写入，跳出循环
                if message_index >= message_length:
                    break
        # 保存图像
        img.save(output_path)
        print("消息已嵌入并保存为:", output_path)
########## End ##########
```

from PIL import Image

def binary_to_text(binary_str):
    #将二进制ASCII码字符串转换为文本，返回文本字符串
    chars = [chr(int(binary_str[i:i+8], 2))
             for i in range(0, len(binary_str), 8)
             if len(binary_str[i:i+8]) == 8]
    return ''.join(chars)
def extract_message(image_path):
    """从RGB图像中提取隐藏的消息"""
    # 加载图
    image = Image.open(image_path)
    pixels = image.load()
    width, height = image.size
    terminator = '1111111111111110'
    # 提取二进制信息
    binary_message = ''
    for x in range(width):
        for y in range(height):
            r, g, b = pixels[x, y]
            bit_r = str(r & 1)
            binary_message += bit_r
            if binary_message.endswith(terminator):
                # 移除终止符
                binary_message = binary_message[:-len(terminator)]
                # 转换为文本并返回
                return binary_to_text(binary_message)
            bit_g = str(g & 1)
            binary_message += bit_g
            if binary_message.endswith(terminator):
                # 移除终止符
                binary_message = binary_message[:-len(terminator)]
                # 转换为文本并返回
                return binary_to_text(binary_message)
            bit_b = str(b & 1)
            binary_message += bit_b
            if binary_message.endswith(terminator):
                # 移除终止符
                binary_message = binary_message[:-len(terminator)]
                # 转换为文本并返回
                return binary_to_text(binary_message)

```python

获取二进制数的最后一位
  使用与&运算，将二进制数和1进行&运算，即可得到二进制数的最低有效位。因为和1&运算，1只有最低位是1，高位全是0，二进制数的高位和1的高位的&运算结构都是0，只有低位运算的结果是二进制的最低位，因为1&1=1,1&0=0。

  示例：
r = 0b11001101
x = r & 1
print(x)  # 输出：1
