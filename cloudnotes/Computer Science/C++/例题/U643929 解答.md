class Account {
private:
    double cash;
    int lot_size;
    double commission_rate;
    double min_commission;
    double stamp_tax;
    
    // 持仓信息
    struct Position {
        int shares;
        double avg_cost;
        string strategy_type;
        int buy_day;
        
        Position() : shares(0), avg_cost(0), strategy_type(""), buy_day(0) {}
        Position(int s, double cost, string type, int day) 
            : shares(s), avg_cost(cost), strategy_type(type), buy_day(day) {}
    };
    
    map<string, Position> positions;
    vector<TradeRecord> trades;
    
    double round_cents(double value) {
        return round(value * 100) / 100;
    }
    
public:
    Account(double init_cash, int lot, double comm_rate, double min_comm, double tax)
        : cash(init_cash), lot_size(lot), commission_rate(comm_rate), 
          min_commission(min_comm), stamp_tax(tax) {}
    
    bool can_buy(double price, int qty) {
        double turnover = price * qty;
        double commission = max(min_commission, commission_rate * turnover);
        double total_cost = turnover + commission;
        return cash >= total_cost;
    }
    
    bool execute_buy(const string& code, double price, int qty, const string& strategy_type, int day) {
        if (!can_buy(price, qty)) return false;
        
        double turnover = price * qty;
        double commission = max(min_commission, commission_rate * turnover);
        double total_cost = round_cents(turnover + commission);
        
        cash = round_cents(cash - total_cost);
        
        if (positions.count(code)) {
            Position& pos = positions[code];
            double total_shares = pos.shares + qty;
            double total_cost_pos = pos.shares * pos.avg_cost + qty * price;
            pos.avg_cost = total_cost_pos / total_shares;
            pos.shares = total_shares;
        } else {
            positions[code] = Position(qty, price, strategy_type, day);
        }
        
        trades.push_back(TradeRecord(code, price, qty, strategy_type, day));
        
        return true;
    }
    
    bool can_sell(const string& code, int qty) {
        return positions.count(code) && positions[code].shares >= qty;
    }
    
    bool execute_sell(const string& code, double price, int qty, int day) {
        if (!can_sell(code, qty)) return false;
        
        double turnover = price * qty;
        double commission = max(min_commission, commission_rate * turnover);
        double tax = stamp_tax * turnover;
        double total_income = round_cents(turnover - commission - tax);
        
        cash = round_cents(cash + total_income);
        
        Position& pos = positions[code];
        pos.shares -= qty;
        
        for (auto& trade : trades) {
            if (trade.code == code && trade.sell_price == 0 && trade.shares == qty) {
                trade.sell_price = price;
                trade.sell_day = day;
                break;
            } else if (trade.code == code && trade.sell_price == 0 && trade.shares > qty) {
                TradeRecord new_trade = trade;
                new_trade.shares = qty;
                new_trade.sell_price = price;
                new_trade.sell_day = day;
                trades.push_back(new_trade);
                
                trade.shares -= qty;
                break;
            }
        }
        
        if (pos.shares == 0) {
            positions.erase(code);
        }
        
        return true;
    }
    
    int get_shares(const string& code) {
        return positions.count(code) ? positions[code].shares : 0;
    }
    
    double get_avg_cost(const string& code) {
        return positions.count(code) ? positions[code].avg_cost : 0;
    }
    
    string get_strategy_type(const string& code) {
        return positions.count(code) ? positions[code].strategy_type : "";
    }
    
    int get_buy_day(const string& code) {
        return positions.count(code) ? positions[code].buy_day : 0;
    }
    
    double get_total_asset(const map<string, double>& last_prices) {
        double total = cash;
        for (const auto& pos : positions) {
            if (last_prices.count(pos.first)) {
                total += pos.second.shares * last_prices.at(pos.first);
            }
        }
        return round_cents(total);
    }
    
    double get_cash() const { return cash; }
};

class TradingStrategy {
private:
    Account& account;
    vector<string> stock_pool;
    map<string, vector<StockData>> history;
    
    // 资金分配
    const double SHORT_RATIO = 0.25;
    const double MEDIUM_RATIO = 0.25;
    const double LONG_RATIO = 0.5;
    
    // 策略参数
    const int SHORT_MA = 5;
    const int MEDIUM_MA1 = 15;
    const int MEDIUM_MA2 = 45;
    const int LONG_MA1 = 5;
    const int LONG_MA2 = 15;
    const int LONG_MA3 = 45;
    
    const double SHORT_BUY_DIP = 0.05;
    const double SHORT_BUY_DIP_MAX = 0.10;
    const double SHORT_SELL_PROFIT = 0.03;
    const double SHORT_STOP_LOSS = 0.15;
    
    const double MEDIUM_PROFIT_TARGET = 0.10;
    const double MEDIUM_STOP_LOSS = 0.15;
    
    const double LONG_PROFIT_TARGET = 0.15;
    const double LONG_STOP_LOSS = 0.10;
    const int LONG_DAILY_INVEST = 100;
    
    double short_capital;
    double medium_capital;
    double long_capital;
    
    double calculate_ma(const vector<double>& prices, int period) {
        if ((int)prices.size() < period) return 0;
        double sum = 0;
        for (int i = (int)prices.size() - period; i < (int)prices.size(); i++) {
            sum += prices[i];
        }
        return sum / period;
    }
    
    double calculate_max_drawdown(const string& code, int days = 45) {
        if ((int)history[code].size() < days) return 0;
        
        double max_price = 0;
        double max_drawdown = 0;
        
        for (int i = (int)history[code].size() - days; i < (int)history[code].size(); i++) {
            if (history[code][i].high > max_price) {
                max_price = history[code][i].high;
            }
            double drawdown = (max_price - history[code][i].low) / max_price;
            if (drawdown > max_drawdown) {
                max_drawdown = drawdown;
            }
        }
        return max_drawdown;
    }
    
    bool is_rising_trend(const string& code) {
        if ((int)history[code].size() < LONG_MA3) return false;
        
        vector<double> closes;
        for (const auto& data : history[code]) {
            closes.push_back(data.close);
        }
        
        double ma5 = calculate_ma(closes, LONG_MA1);
        double ma15 = calculate_ma(closes, LONG_MA2);
        double ma45 = calculate_ma(closes, LONG_MA3);
        
        if (ma5 <= 0 || ma15 <= 0 || ma45 <= 0) return false;
        
        if (ma5 > ma15 && ma15 > ma45) {
            double increase5_15 = (ma5 - ma15) / ma15;
            double increase15_45 = (ma15 - ma45) / ma45;
            
            if (increase5_15 < 0.1 && increase15_45 < 0.2) {
                return true;
            }
        }
        return false;
    }
    
    void update_capital(double last_prices) {
        double total_capital = account.get_total_asset({});
        short_capital = total_capital * SHORT_RATIO;
        medium_capital = total_capital * MEDIUM_RATIO;
        long_capital = total_capital * LONG_RATIO;
    }
    
public:
    TradingStrategy(Account& acc, const vector<string>& pool) 
        : account(acc), stock_pool(pool) {
        double total_capital = 100000;
        short_capital = total_capital * SHORT_RATIO;
        medium_capital = total_capital * MEDIUM_RATIO;
        long_capital = total_capital * LONG_RATIO;
    }
    
    void update_data(const map<string, StockData>& day_data) {
        for (const auto& item : day_data) {
            history[item.first].push_back(item.second);
        }
    }
    
    vector<Order> generate_orders(int current_day, bool is_last_day) {
        vector<Order> orders;
        
        if (is_last_day) {
            for (const auto& code : stock_pool) {
                int shares = account.get_shares(code);
                if (shares > 0) {
                    orders.push_back(Order("SELL", code, shares));
                }
            }
            return orders;
        }
        
        map<string, double> last_prices;
        for (const auto& code : stock_pool) {
            if (!history[code].empty()) {
                last_prices[code] = history[code].back().close;
            }
        }
        update_capital(0);
        
        // 处理卖出信号
        for (const auto& code : stock_pool) {
            int shares = account.get_shares(code);
            if (shares == 0) continue;
            
            const auto& hist = history[code];
            if (hist.empty()) continue;
            
            double current_price = hist.back().close;
            double avg_cost = account.get_avg_cost(code);
            string strategy_type = account.get_strategy_type(code);
            
            if (avg_cost <= 0) continue;
            
            double profit_rate = (current_price - avg_cost) / avg_cost;
            double loss_rate = (avg_cost - current_price) / avg_cost;
            
            if (strategy_type == "SHORT") {
                if (profit_rate >= SHORT_SELL_PROFIT || loss_rate >= SHORT_STOP_LOSS) {
                    orders.push_back(Order("SELL", code, shares));
                }
            }
            else if (strategy_type == "MEDIUM") {
                if (profit_rate >= MEDIUM_PROFIT_TARGET || loss_rate >= MEDIUM_STOP_LOSS) {
                    orders.push_back(Order("SELL", code, shares));
                }
            }
            else if (strategy_type == "LONG") {
                if (profit_rate >= LONG_PROFIT_TARGET || loss_rate >= LONG_STOP_LOSS) {
                    orders.push_back(Order("SELL", code, shares));
                }
            }
        }
        
        // 短线策略
        for (const auto& code : stock_pool) {
            const auto& hist = history[code];
            if ((int)hist.size() < SHORT_MA + 1) continue;
            
            vector<double> closes;
            for (const auto& data : hist) {
                closes.push_back(data.close);
            }
            
            double ma5 = calculate_ma(closes, SHORT_MA);
            double current_price = hist.back().close;
            double prev_price = hist[hist.size() - 2].close;
            
            double daily_drop = (prev_price - current_price) / prev_price;
            
            if (daily_drop >= SHORT_BUY_DIP && daily_drop <= SHORT_BUY_DIP_MAX) {
                if (current_price < ma5) {
                    double current_short_used = 0;
                    for (const auto& c : stock_pool) {
                        if (account.get_strategy_type(c) == "SHORT") {
                            current_short_used += account.get_shares(c) * current_price;
                        }
                    }
                    
                    if (current_short_used < short_capital * 0.8) {
                        int max_shares = (int)((short_capital - current_short_used) / current_price / 100) * 100;
                        if (max_shares >= 100) {
                            int qty = min(500, max_shares);
                            if (account.can_buy(current_price, qty)) {
                                orders.push_back(Order("BUY", code, qty));
                            }
                        }
                    }
                }
            }
        }
        
        // 中线策略
        for (const auto& code : stock_pool) {
            const auto& hist = history[code];
            if ((int)hist.size() < MEDIUM_MA2) continue;
            
            vector<double> closes;
            for (const auto& data : hist) {
                closes.push_back(data.close);
            }
            
            double ma15 = calculate_ma(closes, MEDIUM_MA1);
            double ma45 = calculate_ma(closes, MEDIUM_MA2);
            double current_price = hist.back().close;
            
            if (current_price < ma15 && ma15 > ma45) {
                double current_medium_used = 0;
                for (const auto& c : stock_pool) {
                    if (account.get_strategy_type(c) == "MEDIUM") {
                        current_medium_used += account.get_shares(c) * current_price;
                    }
                }
                
                if (current_medium_used < medium_capital * 0.8) {
                    int max_shares = (int)((medium_capital - current_medium_used) / current_price / 100) * 100;
                    if (max_shares >= 100) {
                        int qty = min(300, max_shares);
                        if (account.can_buy(current_price, qty)) {
                            orders.push_back(Order("BUY", code, qty));
                        }
                    }
                }
            }
        }
        
        // 长线策略
        for (const auto& code : stock_pool) {
            const auto& hist = history[code];
            if ((int)hist.size() < LONG_MA3) continue;
            
            double max_drawdown = calculate_max_drawdown(code);
            if (max_drawdown > 0.15) continue;
            
            if (!is_rising_trend(code)) continue;
            
            double current_price = hist.back().close;
            
            double current_long_used = 0;
            for (const auto& c : stock_pool) {
                if (account.get_strategy_type(c) == "LONG") {
                    current_long_used += account.get_shares(c) * current_price;
                }
            }
            
            if (current_long_used < long_capital) {
                if (account.can_buy(current_price, LONG_DAILY_INVEST)) {
                    orders.push_back(Order("BUY", code, LONG_DAILY_INVEST));
                }
            }
        }
        
        return orders;
    }
};

                new_trade.sell_price = price;
                new_trade.sell_day = day;
                trades.push_back(new_trade);
                
                trade.shares -= qty;
                break;
            }
        }
        
        if (pos.shares == 0) {
            positions.erase(code);
        }
        
        return true;
    }
    
    int get_shares(const string& code) {
        return positions.count(code) ? positions[code].shares : 0;
    }
    
    double get_avg_cost(const string& code) {
        return positions.count(code) ? positions[code].avg_cost : 0;
    }
    
    string get_strategy_type(const string& code) {
        return positions.count(code) ? positions[code].strategy_type : "";
    }
    
    int get_buy_day(const string& code) {
        return positions.count(code) ? positions[code].buy_day : 0;
    }
    
    double get_last_price(const string& code) {
        return positions.count(code) ? positions[code].last_price : 0;
    }
    
    double get_total_asset(const map<string, double>& last_prices) {
        double total = cash;
        for (const auto& pos : positions) {
            if (last_prices.count(pos.first)) {
                total += pos.second.shares * last_prices.at(pos.first);
            }
        }
        return round_cents(total);
    }
    
    double get_cash() const { return cash; }
};

// 市场状态类 - 用于整体市场分析
class MarketState {
private:
    vector<double> market_returns;
    vector<double> market_volumes;
    
public:
    void update(double avg_return, double total_volume) {
        market_returns.push_back(avg_return);
        market_volumes.push_back(total_volume);
        if (market_returns.size() > 20) {
            market_returns.erase(market_returns.begin());
            market_volumes.erase(market_volumes.begin());
        }
    }
    
    double get_market_trend() {
        if (market_returns.size() < 5) return 0;
        double sum = 0;
        for (int i = (int)market_returns.size() - 5; i < (int)market_returns.size(); i++) {
            sum += market_returns[i];
        }
        return sum / 5;  // 5日平均涨跌幅
    }
    
    double get_market_volume_ratio() {
        if (market_volumes.size() < 10) return 1.0;
        double current = market_volumes.back();
        double sum = 0;
        for (int i = 0; i < (int)market_volumes.size() - 1; i++) {
            sum += market_volumes[i];
        }
        double avg = sum / (market_volumes.size() - 1);
        return current / avg;
    }
    
    bool is_bull_market() {
        return get_market_trend() > 0.2;  // 平均涨幅>0.2%
    }
    
    bool is_bear_market() {
        return get_market_trend() < -0.2;  // 平均跌幅>0.2%
    }
};

// 多因子评分策略类
class MultiFactorStrategy {
private:
    Account& account;
    vector<string> stock_pool;
    map<string, vector<StockData>> history;
    
    // 策略状态机
    enum StrategyState {
        STATE_IDLE,           // 空闲，可产生信号
        STATE_PENDING_BUY,    // 有买入委托等待成交
        STATE_PENDING_SELL,   // 有卖出委托等待成交
        STATE_COOLDOWN        // 冷却中，不产生信号
    };
    
    map<string, StrategyState> stock_states;
    map<string, int> state_days;
    const int COOLDOWN_DAYS = 3;  // 交易失败后的冷却天数
    
    // 资金分配
    const double SHORT_RATIO = 0.25;
    const double MEDIUM_RATIO = 0.25;
    const double LONG_RATIO = 0.5;
    
    // 多因子权重
    struct FactorWeights {
        double price_trend;      // 价格趋势因子
        double volume_factor;    // 成交量因子
        double volatility;       // 波动率因子
        double momentum;         // 动量因子
        double mean_reversion;   // 均值回归因子
        
        FactorWeights() : price_trend(0.3), volume_factor(0.2), 
                          volatility(0.15), momentum(0.2), mean_reversion(0.15) {}
    } weights;
    
    // 策略参数
    const int SHORT_MA = 5;
    const int MEDIUM_MA1 = 15;
    const int MEDIUM_MA2 = 45;
    const int LONG_MA1 = 5;
    const int LONG_MA2 = 15;
    const int LONG_MA3 = 45;
    const int VOLUME_MA = 10;
    
    // 买入阈值（根据市场状态动态调整）
    double SHORT_BUY_THRESHOLD = 2.5;
    double MEDIUM_BUY_THRESHOLD = 2.0;
    double LONG_BUY_THRESHOLD = 3.0;
    
    // 卖出阈值
    const double SHORT_PROFIT_TARGET = 0.03;
    const double SHORT_STOP_LOSS = 0.15;
    const double MEDIUM_PROFIT_TARGET = 0.10;
    const double MEDIUM_STOP_LOSS = 0.15;
    const double LONG_PROFIT_TARGET = 0.15;
    const double LONG_STOP_LOSS = 0.10;
    
    // 动态调整的仓位限制
    double short_capital;
    double medium_capital;
    double long_capital;
    
    // 市场状态
    MarketState market;
    
    // 辅助函数
    double calculate_ma(const vector<double>& prices, int period) {
        if ((int)prices.size() < period) return 0;
        double sum = 0;
        for (int i = (int)prices.size() - period; i < (int)prices.size(); i++) {
            sum += prices[i];
        }
        return sum / period;
    }
    
    double calculate_ema(const vector<double>& prices, int period) {
        if ((int)prices.size() < period) return prices.back();
        double multiplier = 2.0 / (period + 1);
        double ema = prices[0];
        for (int i = 1; i < (int)prices.size(); i++) {
            ema = (prices[i] - ema) * multiplier + ema;
        }
        return ema;
    }
    
    double calculate_rsi(const vector<double>& prices, int period = 14) {
        if ((int)prices.size() < period + 1) return 50;
        
        double gain = 0, loss = 0;
        for (int i = (int)prices.size() - period; i < (int)prices.size(); i++) {
            double change = prices[i] - prices[i-1];
            if (change > 0) gain += change;
            else loss -= change;
        }
        
        double avg_gain = gain / period;
        double avg_loss = loss / period;
        
        if (avg_loss == 0) return 100;
        double rs = avg_gain / avg_loss;
        return 100 - (100 / (1 + rs));
    }
    
    double calculate_max_drawdown(const string& code, int days = 45) {
        if ((int)history[code].size() < days) return 0;
        
        double max_price = 0;
        double max_drawdown = 0;
        
        for (int i = (int)history[code].size() - days; i < (int)history[code].size(); i++) {
            if (history[code][i].high > max_price) {
                max_price = history[code][i].high;
            }
            double drawdown = (max_price - history[code][i].low) / max_price;
            if (drawdown > max_drawdown) {
                max_drawdown = drawdown;
            }
        }
        return max_drawdown;
    }
    
    double calculate_volume_ratio(const string& code) {
        const auto& hist = history[code];
        if ((int)hist.size() < VOLUME_MA + 1) return 1.0;
        
        long long current_volume = hist.back().volume;
        long long sum = 0;
        for (int i = (int)hist.size() - VOLUME_MA - 1; i < (int)hist.size() - 1; i++) {
            sum += hist[i].volume;
        }
        double avg_volume = sum / VOLUME_MA;
        
        return current_volume / avg_volume;
    }
    
    double calculate_bollinger_position(const string& code, int period = 20) {
        const auto& hist = history[code];
        if ((int)hist.size() < period) return 0;
        
        vector<double> closes;
        for (int i = (int)hist.size() - period; i < (int)hist.size(); i++) {
            closes.push_back(hist[i].close);
        }
        
        double ma = calculate_ma(closes, period);
        double sum = 0;
        for (double c : closes) {
            sum += (c - ma) * (c - ma);
        }
        double std_dev = sqrt(sum / period);
        
        double current = hist.back().close;
        double upper = ma + 2 * std_dev;
        double lower = ma - 2 * std_dev;
        
        if (current >= upper) return 1.0;
        if (current <= lower) return -1.0;
        return (current - ma) / (2 * std_dev);  // -1 到 1 之间的位置
    }
    
    // 多因子评分系统
    double score_short_term(const string& code) {
        const auto& hist = history[code];
        if ((int)hist.size() < SHORT_MA + 5) return 0;
        
        vector<double> closes;
        for (const auto& d : hist) closes.push_back(d.close);
        
        double score = 0;
        
        // 1. 价格趋势因子 (30%)
        double ma5 = calculate_ma(closes, SHORT_MA);
        double current = hist.back().close;
        double price_trend = (current - ma5) / ma5 * 100;  // 偏离度百分比
        score += weights.price_trend * (price_trend > 0 ? -price_trend : -price_trend * 0.5);
        
        // 2. 成交量因子 (20%)
        double volume_ratio = calculate_volume_ratio(code);
        if (volume_ratio > 1.5 && hist.back().pct_chg < -2) {
            score += weights.volume_factor * 2;  // 放量下跌是买入信号
        }
        
        // 3. 波动率因子 (15%)
        double amplitude = hist.back().amplitude;
        double avg_amplitude = 0;
        for (int i = (int)hist.size() - 10; i < (int)hist.size(); i++) {
            avg_amplitude += hist[i].amplitude;
        }
        avg_amplitude /= 10;
        if (amplitude > avg_amplitude * 1.5) {
            score += weights.volatility * 1.5;  // 高波动可能带来机会
        }
        
        // 4. 动量因子 (20%)
        double rsi = calculate_rsi(closes, 14);
        if (rsi < 30) score += weights.momentum * 2;  // 超卖
        else if (rsi > 70) score -= weights.momentum * 1;  // 超买
        
        // 5. 均值回归因子 (15%)
        double daily_drop = hist.size() >= 2 ? 
            (hist[hist.size()-2].close - current) / hist[hist.size()-2].close : 0;
        if (daily_drop > 0.05 && daily_drop < 0.10) {
            score += weights.mean_reversion * 3;  // 5-10%跌幅是强信号
        }
        
        return score;
    }
    
    double score_medium_term(const string& code) {
        const auto& hist = history[code];
        if ((int)hist.size() < MEDIUM_MA2) return 0;
        
        vector<double> closes;
        for (const auto& d : hist) closes.push_back(d.close);
        
        double score = 0;
        
        // 1. 价格趋势因子
        double ma15 = calculate_ma(closes, MEDIUM_MA1);
        double ma45 = calculate_ma(closes, MEDIUM_MA2);
        double current = hist.back().close;
        
        if (current < ma15 && ma15 > ma45) {
            score += weights.price_trend * 3;  // 金叉回调是买入机会
        }
        
        // 2. 成交量因子
        double volume_ratio = calculate_volume_ratio(code);
        if (volume_ratio > 1.2 && hist.back().pct_chg > 1) {
            score += weights.volume_factor * 1.5;  // 温和放量上涨
        }
        
        // 3. 波动率因子
        double max_drawdown = calculate_max_drawdown(code, 20);
        if (max_drawdown < 0.1) {
            score += weights.volatility * 1;  // 低回撤股票加分
        }
        
        // 4. 动量因子
        double rsi = calculate_rsi(closes, 14);
        if (rsi > 40 && rsi < 60) {
            score += weights.momentum * 1;  // 中性区域
        }
        
        // 5. 均值回归因子
        double bollinger_pos = calculate_bollinger_position(code);
        if (bollinger_pos < -0.5) {
            score += weights.mean_reversion * 2;  // 下轨附近
        }
        
        return score;
    }
    
    double score_long_term(const string& code) {
        const auto& hist = history[code];
        if ((int)hist.size() < LONG_MA3) return 0;
        
        vector<double> closes;
        for (const auto& d : hist) closes.push_back(d.close);
        
        double score = 0;
        
        // 1. 价格趋势因子
        double ma5 = calculate_ma(closes, LONG_MA1);
        double ma15 = calculate_ma(closes, LONG_MA2);
        double ma45 = calculate_ma(closes, LONG_MA3);
        
        if (ma5 > ma15 && ma15 > ma45) {
            score += weights.price_trend * 3;  // 多头排列
        }
        
        // 2. 成交量因子
        long long sum_volume = 0;
        for (int i = (int)hist.size() - 20; i < (int)hist.size(); i++) {
            sum_volume += hist[i].volume;
        }
        double avg_volume = sum_volume / 20.0;
        double volume_stability = 0;
        for (int i = (int)hist.size() - 20; i < (int)hist.size(); i++) {
            volume_stability += abs(hist[i].volume - avg_volume) / avg_volume;
        }
        volume_stability /= 20;
        if (volume_stability < 0.3) {
            score += weights.volume_factor * 1.5;  // 成交量稳定
        }
        
        // 3. 波动率因子
        double max_drawdown = calculate_max_drawdown(code, LONG_MA3);
        if (max_drawdown < 0.15) {
            score += weights.volatility * 2;  // 回撤小加分
        } else {
            score -= weights.volatility * 2;  // 回撤大减分
        }
        
        // 4. 动量因子
        double momentum_score = 0;
        for (int i = (int)closes.size() - 20; i < (int)closes.size(); i++) {
            momentum_score += closes[i] > closes[i-1] ? 1 : -1;
        }
        score += weights.momentum * (momentum_score / 10);  // 动量强度
        
        // 5. 均值回归因子
        double ma_increase = (ma5 - ma45) / ma45 * 100;
        if (ma_increase > 5 && ma_increase < 20) {
            score += weights.mean_reversion * 2;  // 合理涨幅
        }
        
        return score;
    }
    
    void update_capital() {
        double total_capital = account.get_cash();
        map<string, double> last_prices;
        for (const auto& code : stock_pool) {
            if (!history[code].empty()) {
                last_prices[code] = history[code].back().close;
            }
        }
        total_capital = account.get_total_asset(last_prices);
        
        // 根据市场状态动态调整资金分配
        if (market.is_bull_market()) {
            // 牛市增加长线比例
            short_capital = total_capital * 0.2;
            medium_capital = total_capital * 0.2;
            long_capital = total_capital * 0.6;
        } else if (market.is_bear_market()) {
            // 熊市减少长线，增加现金
            short_capital = total_capital * 0.15;
            medium_capital = total_capital * 0.15;
            long_capital = total_capital * 0.3;
        } else {
            short_capital = total_capital * SHORT_RATIO;
            medium_capital = total_capital * MEDIUM_RATIO;
            long_capital = total_capital * LONG_RATIO;
        }
        
        // 动态调整阈值
        if (market.get_market_volume_ratio() > 1.5) {
            SHORT_BUY_THRESHOLD = 2.0;  // 放量市场降低阈值
        } else {
            SHORT_BUY_THRESHOLD = 2.5;
        }
    }
    
public:
    MultiFactorStrategy(Account& acc, const vector<string>& pool) 
        : account(acc), stock_pool(pool) {
        double total_capital = 100000;
        short_capital = total_capital * SHORT_RATIO;
        medium_capital = total_capital * MEDIUM_RATIO;
        long_capital = total_capital * LONG_RATIO;
        
        // 初始化所有股票状态为空闲
        for (const auto& code : pool) {
            stock_states[code] = STATE_IDLE;
            state_days[code] = 0;
        }
    }
    
    void update_data(const map<string, StockData>& day_data) {
        // 更新历史数据
        for (const auto& item : day_data) {
            history[item.first].push_back(item.second);
        }
        
        // 更新市场状态
        double total_return = 0;
        long long total_volume = 0;
        int valid_count = 0;
        for (const auto& item : day_data) {
            if (!history[item.first].empty() && history[item.first].size() > 1) {
                total_return += item.second.pct_chg;
                total_volume += item.second.volume;
                valid_count++;
            }
        }
        if (valid_count > 0) {
            market.update(total_return / valid_count, total_volume);
        }
        
        // 更新状态机的冷却天数
        for (auto& state_pair : state_days) {
            if (state_pair.second > 0) {
                state_pair.second--;
                if (state_pair.second == 0 && stock_states[state_pair.first] == STATE_COOLDOWN) {
                    stock_states[state_pair.first] = STATE_IDLE;
                }
            }
        }
    }
    
    void update_pending_state(const string& code, bool is_buy) {
        // 提交委托后更新状态
        stock_states[code] = is_buy ? STATE_PENDING_BUY : STATE_PENDING_SELL;
    }
    
    void update_settled_state(const string& code, bool success) {
        // 委托执行后更新状态
        if (success) {
            stock_states[code] = STATE_IDLE;  // 成交成功，回到空闲
        } else {
            stock_states[code] = STATE_COOLDOWN;  // 失败，进入冷却
            state_days[code] = COOLDOWN_DAYS;
        }
    }
    
    vector<Order> generate_orders(int current_day, bool is_last_day) {
        vector<Order> orders;
        
        if (is_last_day) {
            // 最后一天清仓所有股票
            for (const auto& code : stock_pool) {
                int shares = account.get_shares(code);
                if (shares > 0) {
                    orders.push_back(Order("SELL", code, shares));
                }
            }
            return orders;
        }
        
        // 更新可用资金
        update_capital();
        
        // 处理卖出信号（止盈止损）
        for (const auto& code : stock_pool) {
            int shares = account.get_shares(code);
            if (shares == 0) continue;
            
            const auto& hist = history[code];
            if (hist.empty()) continue;
            
            double current_price = hist.back().close;
            double avg_cost = account.get_avg_cost(code);
            string strategy_type = account.get_strategy_type(code);
            
            if (avg_cost <= 0) continue;
            
            double profit_rate = (current_price - avg_cost) / avg_cost;
            double loss_rate = (avg_cost - current_price) / avg_cost;
            
            bool need_sell = false;
            
            if (strategy_type == "SHORT") {
                if (profit_rate >= SHORT_PROFIT_TARGET || loss_rate >= SHORT_STOP_LOSS) {
                    need_sell = true;
                }
            } else if (strategy_type == "MEDIUM") {
                if (profit_rate >= MEDIUM_PROFIT_TARGET || loss_rate >= MEDIUM_STOP_LOSS) {
                    need_sell = true;
                }
            } else if (strategy_type == "LONG") {
                if (profit_rate >= LONG_PROFIT_TARGET || loss_rate >= LONG_STOP_LOSS) {
                    need_sell = true;
                }
            }
            
            if (need_sell && stock_states[code] == STATE_IDLE) {
                orders.push_back(Order("SELL", code, shares));
                stock_states[code] = STATE_PENDING_SELL;
            }
        }
        
        // 买入信号 - 多因子评分
        for (const auto& code : stock_pool) {
            // 状态检查：只有空闲状态的股票才考虑买入
            if (stock_states[code] != STATE_IDLE) continue;
            
            const auto& hist = history[code];
            if ((int)hist.size() < LONG_MA3) continue;
            
            double current_price = hist.back().close;
            
            // 计算各策略评分
            double short_score = score_short_term(code);
            double medium_score = score_medium_term(code);
            double long_score = score_long_term(code);
            
            // 计算各策略已用资金
            double current_short_used = 0;
            double current_medium_used = 0;
            double current_long_used = 0;
            
            for (const auto& c : stock_pool) {
                string type = account.get_strategy_type(c);
                int shares = account.get_shares(c);
                if (shares > 0 && !history[c].empty()) {
                    double price = history[c].back().close;
                    if (type == "SHORT") current_short_used += shares * price;
                    else if (type == "MEDIUM") current_medium_used += shares * price;
                    else if (type == "LONG") current_long_used += shares * price;
                }
            }
            
            // 短线买入
            if (short_score >= SHORT_BUY_THRESHOLD && current_short_used < short_capital * 0.8) {
                int max_shares = (int)((short_capital - current_short_used) / current_price / 100) * 100;
                if (max_shares >= 100) {
                    int qty = min(300, max_shares);
                    if (account.can_buy(current_price, qty)) {
                        orders.push_back(Order("BUY", code, qty));
                        stock_states[code] = STATE_PENDING_BUY;
                        continue;  // 一股一策，买了一个就不再买其他策略
                    }
                }
            }
            
            // 中线买入
            if (medium_score >= MEDIUM_BUY_THRESHOLD && current_medium_used < medium_capital * 0.8) {
                int max_shares = (int)((medium_capital - current_medium_used) / current_price / 100) * 100;
                if (max_shares >= 100) {
                    int qty = min(200, max_shares);
                    if (account.can_buy(current_price, qty)) {
                        orders.push_back(Order("BUY", code, qty));
                        stock_states[code] = STATE_PENDING_BUY;
                        continue;
                    }
                }
            }
            
            // 长线买入
            if (long_score >= LONG_BUY_THRESHOLD && current_long_used < long_capital * 0.8) {
                int max_shares = (int)((long_capital - current_long_used) / current_price / 100) * 100;
                if (max_shares >= 100) {
                    int qty = min(200, max_shares);
                    if (account.can_buy(current_price, qty)) {
                        orders.push_back(Order("BUY", code, qty));
                        stock_states[code] = STATE_PENDING_BUY;
                        continue;
                    }
                }
            }
        }
        
        return orders;
    }
    
    // 获取股票状态（供主程序使用）
    StrategyState get_state(const string& code) {
        return stock_states[code];
    }
};

        Account account(m, L, alpha, com_min, beta);
        MultiFactorStrategy strategy(account, stock_pool);
        strategy.update_data(day0_data);
        
        map<string, StockData> current_data = day0_data;
        
        for (int day = 1; day <= D; day++) {
            bool is_last_day = (day == D);
            vector<Order> orders = strategy.generate_orders(day, is_last_day);
            
            // 检查委托数量上限
            if ((int)orders.size() > K_max) {
                orders.resize(K_max);
            }
            
            // 输出委托
            for (const auto& order : orders) {
                cout << order.type << " " << order.code << " " << order.qty << "\n";
                // 更新策略状态
                strategy.update_pending_state(order.code, order.type == "BUY");
            }
            cout << "DONE\n";
            cout.flush();
            
            // 读取下一日行情
            map<string, StockData> next_day_data;
            for (int i = 0; i < n; i++) {
                if (!getline(cin, line)) break;
                
                if (line.find("FINISH") != string::npos) {
                    return 0;
                }
                
                if (line.empty()) {
                    i--;
                    continue;
                }
                
                StockData data(line);
                next_day_data[data.code] = data;
            }
            
            if (!next_day_data.empty()) {
                // 在实际执行后，更新状态
                for (const auto& order : orders) {
                    // 简化处理：假设委托都成功执行
                    // 在实际环境中，应该根据执行结果更新
                    strategy.update_settled_state(order.code, true);
                }
                
                current_data = next_day_data;
                strategy.update_data(current_data);
            }
        }
        
        // 第D+1组委托：只输出DONE
        cout << "DONE\n";
        cout.flush();
        
        while (getline(cin, line)) {
            if (line.find("FINISH") != string::npos) {
                break;
            }
        }
        
    } catch (const exception& e) {
        cerr << "Error: " << e.what() << endl;
        for (int i = 0; i < 250; i++) {
            cout << "DONE\n";
            cout.flush();
        }
    }
    
    return 0;
}
