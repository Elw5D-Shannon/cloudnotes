## 逐行输入

```cpp
#include <iostream>
#include <sstream>
#include <string>
#include <vector>
using namespace std;
int main() {
    string line;
    vector<int> numbers;
    // 读取每一行
    while (getline(cin, line)) {
        if (line.empty()) continue;
        
        // 使用字符串流解析当前行
        istringstream iss(line);
        int num;
        numbers.clear();
        
        cout << "当前行数字: ";
        while (iss >> num) {
            numbers.push_back(num);
            cout << num << " ";
        }
        cout << endl;
        
        // 遍历当前行的数字
        for (int n : numbers) {
            cout << "处理数字: " << n << endl;
        }
    }
    return 0;
} 
```
