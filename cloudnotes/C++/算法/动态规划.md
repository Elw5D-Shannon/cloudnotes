### 什么是动态规划？

动态规划（Dynamic Programming，DP）是一种通过将复杂问题分解为更简单的子问题来解决的算法思想。它通常用于求解具有**最优子结构**和**重叠子问题**特性的问题。

#### 核心特征：

1. **最优子结构** - 问题的最优解包含子问题的最优解
    
2. **重叠子问题** - 子问题会被重复计算多次
    
3. **状态转移方程** - 定义子问题之间的关系
    
4. **记忆化存储** - 保存已解决的子问题结果
    

### 动态规划的两种实现方式

#### 1. 自顶向下（Top-down） - 递归 + 备忘录
```cpp

#include <vector>
#include <unordered_map>
using namespace std;
// 示例：斐波那契数列
int fib(int n, unordered_map<int, int>& memo) {
    if (n <= 1) return n;
    
    // 如果已经计算过，直接返回
    if (memo.find(n) != memo.end()) {
        return memo[n];
    }
    
    // 计算并存储结果
    memo[n] = fib(n - 1, memo) + fib(n - 2, memo);
    return memo[n];
}

#### 2. 自底向上（Bottom-up） - 迭代 + 表格

cpp

// 示例：斐波那契数列 - 自底向上
int fib(int n) {
    if (n <= 1) return n;
    
    vector<int> dp(n + 1);
    dp[0] = 0;
    dp[1] = 1;
    
    for (int i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    
    return dp[n];
}
// 空间优化版本
int fibOptimized(int n) {
    if (n <= 1) return n;
    
    int prev2 = 0, prev1 = 1;
    for (int i = 2; i <= n; i++) {
        int curr = prev1 + prev2;
        prev2 = prev1;
        prev1 = curr;
    }
    return prev1;
}

```

### 经典动态规划问题示例

#### 1. 爬楼梯问题
```cpp

/**
 * 问题：每次可以爬1或2个台阶，爬到n阶有多少种方法？
 */
class Solution {
public:
    int climbStairs(int n) {
        if (n <= 2) return n;
        
        vector<int> dp(n + 1);
        dp[1] = 1;
        dp[2] = 2;
        
        for (int i = 3; i <= n; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];  // 状态转移方程
        }
        
        return dp[n];
    }
};

```

#### 2. 最大子数组和（Kadane算法）
```cpp

/**
 * 问题：找到具有最大和的连续子数组
 */
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int n = nums.size();
        vector<int> dp(n);  // dp[i]表示以i结尾的最大子数组和
        dp[0] = nums[0];
        int maxSum = dp[0];
        
        for (int i = 1; i < n; i++) {
            // 要么从当前元素开始，要么加入之前的子数组
            dp[i] = max(nums[i], dp[i - 1] + nums[i]);
            maxSum = max(maxSum, dp[i]);
        }
        
        return maxSum;
    }
};

```

#### 3. 0-1背包问题
```cpp

/**
 * 问题：给定物品重量和价值，背包容量W，求最大价值
 */
class Solution {
public:
    int knapsack(vector<int>& weights, vector<int>& values, int W) {
        int n = weights.size();
        vector<vector<int>> dp(n + 1, vector<int>(W + 1, 0));
        
        for (int i = 1; i <= n; i++) {
            for (int w = 1; w <= W; w++) {
                if (weights[i - 1] <= w) {
                    // 选择放入或不放入当前物品
                    dp[i][w] = max(
                        dp[i - 1][w],  // 不放入
                        dp[i - 1][w - weights[i - 1]] + values[i - 1]  // 放入
                    );
                } else {
                    dp[i][w] = dp[i - 1][w];
                }
            }
        }
        
        return dp[n][W];
    }
    
    // 空间优化版本
    int knapsackOptimized(vector<int>& weights, vector<int>& values, int W) {
        int n = weights.size();
        vector<int> dp(W + 1, 0);
        
        for (int i = 0; i < n; i++) {
            // 注意：必须从右向左遍历，避免重复使用同一物品
            for (int w = W; w >= weights[i]; w--) {
                dp[w] = max(dp[w], dp[w - weights[i]] + values[i]);
            }
        }
        
        return dp[W];
    }
};

```

#### 4. 最长公共子序列（LCS）
```cpp

/**
 * 问题：求两个字符串的最长公共子序列长度
 */
class Solution {
public:
    int longestCommonSubsequence(string text1, string text2) {
        int m = text1.length(), n = text2.length();
        vector<vector<int>> dp(m + 1, vector<int>(n + 1, 0));
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (text1[i - 1] == text2[j - 1]) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                } else {
                    dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }
        
        return dp[m][n];
    }
};

```

### 动态规划解题步骤

1. **定义状态** - 明确dp数组/函数的含义
    
2. **确定状态转移方程** - 找出子问题之间的关系
    
3. **初始化** - 设置base case
    
4. **确定遍历顺序** - 保证计算当前状态时所需的状态已计算
    
5. **返回结果** - 根据定义返回最终答案
    

### 常见动态规划类型

|类型|特点|经典问题|
|---|---|---|
|线性DP|一维状态，线性递推|爬楼梯、最大子数组和|
|区间DP|状态表示区间|矩阵链乘、回文子串|
|背包DP|有限制条件的选择问题|0-1背包、完全背包|
|树形DP|在树结构上DP|树的直径、树的最大独立集|
|状态压缩DP|用二进制表示状态|旅行商问题|

### 总结

动态规划是一种强大的算法思想，关键在于：

- 识别问题是否具有最优子结构和重叠子问题
    
- 正确设计状态和状态转移方程
    
- 选择合适的实现方式（自顶向下或自底向上）
    
- 考虑是否需要空间优化
    

通过多练习经典问题，可以逐渐培养动态规划的思维模式。
